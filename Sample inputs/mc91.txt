environment
mcNinetyOne : int->int->bool

program
# The name of the top-level relational variable was originally mc91.
# However, because digits cannot be in variable names, the variable name is replaced with 'mcNinetyOne'.
mcNinetyOne := \x: int. \r: int. (
  (x > 100 && r = x-10)
  ||
  (x <= 100 && (E y:int.(mcNinetyOne (x + 11) y) && mcNinetyOne y r))
);

goal
E x: int. E y:int. x <= 101 && (y < 91 || y > 91) && mcNinetyOne x y
